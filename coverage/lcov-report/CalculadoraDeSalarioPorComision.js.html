class CalculadoraDeSalarioPorComision{
    constructor(salarioBase,porcentajeDeComision,kardex, tarjetaDeVentas){
        this.salarioBase = salarioBase;
        this.porcentajeDeComision = porcentajeDeComision;
        this.kardex=kardex;
        this.tarjetasDeVentas = tarjetaDeVentas;
    }
 
    calcularSalario(){
        let ventasTotales=this.calcularVentasTotales();
        return (this.salarioBase-this.kardex)+(this.porcentajeDeComision*ventasTotales);
    }
 
    calcularVentasTotales(){
        let total=0;
        for (let contador=0; contador<this.tarjetasDeVentas.length; contador++){
            total=total+this.tarjetasDeVentas[contador].obtenerMontoDeVentas();
        }
        return total;
    }
 
}
 
export default CalculadoraDeSalarioPorComision;
